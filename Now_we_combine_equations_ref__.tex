Now we combine equations \ref{eqn:dbetahat} and \ref{eqn:d2betahat} in the following way:
\begin{equation}
\frac{d^2 \betahat_{i j}}{d X_j^2} + \frac{2}{X_j} \frac{d \betahat_{i j}}{d X_j} = \frac{d^2 \beta_{i j}}{d X_j^2} -  \betahat_{i j} \frac{d \kappa}{d X_j} \left. \frac{d \beta(z, z_j)}{d X_j}  \right|_{z \rightarrow z_j}  - \int_{X_i}^{X_j} \betahat(z_i, z) \frac{d^2 \beta(z, z_j)}{d X_j^2} \frac{d \kappa}{d X} d X.
\end{equation}
